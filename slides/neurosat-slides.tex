% ------------------------------------------------------------------ SAT problem
\begin{frame}{The SAT problem}
  \begin{itemize}
    \item \textbf{SAT}: is there an assignment satisfying a Boolean (CNF) formula?
          First problem proven NP-complete; core of verification, planning, synthesis.
    \item Modern solvers are \textbf{CDCL}: decide a variable, unit-propagate, on a
          conflict learn a clause and backtrack.
    \item The \textbf{branching heuristic} (e.g.\ VSIDS) picks the decision variable
          --- and makes poor choices during its warm-up phase.
    \item \emph{This project}: learn the branching heuristic with two neuro-symbolic
          methods, modernise them to PyTorch, reproduce, and study \textbf{where
          graph attention helps}.
  \end{itemize}
\end{frame}

\begin{frame}{Contributions}
  \begin{itemize}
    \item Port both methods from \textbf{TensorFlow 1.x + GSL} to one self-contained
          \textbf{PyTorch / PyTorch-Geometric} stack (latest \texttt{torch}, \texttt{numpy}$\geq$2,
          \texttt{gymnasium}; no \texttt{torch-scatter}/\texttt{sparse}, no GSL).
    \item \textbf{Reproduce the published Graph-Q-SAT results exactly} (semantic
          non-regression oracle).
    \item \textbf{GAT-Q-SAT}: an attention-augmented variant, with a clean
          attention-on/off study showing \emph{where} attention helps.
  \end{itemize}
\end{frame}

  Selsam et al., \emph{Learning a SAT Solver from Single-Bit Supervision}.}

\begin{frame}{NeuroSAT (survey)}
  \begin{itemize}
    \item \textbf{Goal}: can a neural network learn to solve SAT?
    \item An MPNN trained \emph{only} as a SAT/UNSAT classifier (one-bit supervision)
          nonetheless learns to find satisfying assignments.
    \item Introduces the \textbf{graph view} of a CNF (variable/clause nodes) that
          underpins Graph-Q-SAT --- our second method.
  \end{itemize}
\end{frame}

% ================================================================== AlphaZeroSAT
\sectiondivider{AlphaZeroSAT}{%
  \begin{itemize}
    \item AlphaZeroSAT
    \item Background \begin{itemize}\item Monte Carlo Tree Search (MCTS)\end{itemize}
    \item Model \item Training \item Results
  \end{itemize}}{(ISPR + RL -- survey with coding)\\[2mm]
  Wang \& Rompf, \emph{From Gameplay to Symbolic Reasoning \dots\ Alpha(Go) Zero}.}

\begin{frame}{AlphaZeroSAT --- model \& training}
  \begin{itemize}
    \item State: CNF as a fixed-size \textbf{clauses\,$\times$\,variables} matrix; a
          \textbf{CNN} with policy head $\pi$ and value head $v\in[-1,1]$.
    \item MiniSat extended into a \emph{game} env supporting MCTS \textbf{simulations}.
    \item Self-play + supervised training with the AlphaZero loss
          $-\sum_a \pi^{\text{MCTS}}_a\log\pi_a + (z-v)^2 + c\lVert\theta\rVert^2$.
    \item \textbf{This work}: ported to PyTorch; \textbf{GSL removed} (Dirichlet noise
          via C++11 \texttt{<random>}); trains end-to-end on \texttt{uf20-91} (CPU/GPU).
    \item \emph{Limitation}: the fixed-size CNN \textbf{cannot generalise across sizes}
          --- the motivation for graphs.
  \end{itemize}
\end{frame}

\begin{frame}{Graph-Q-SAT \& GAT-Q-SAT --- model}
  \begin{itemize}
    \item State: \textbf{bipartite graph} (variable / clause nodes); edge features =
          literal polarity. $Q$-function = \textbf{encode--process--decode} graph net; DQN.
    \item Action: $\arg\max$ $Q$ over variable nodes. Metric: \textbf{MRIR} (MiniSat
          iters / model iters; $>1$ is better than MiniSat).
    \item \textbf{GAT-Q-SAT (this work)}: inject edge-aware, multi-head
          \texttt{GATConv} attention into the core block --- weight the most relevant
          neighbouring clauses/variables.
    \item Port: dropped \texttt{torch-scatter}/\texttt{sparse}; gym-robust env;
          version-agnostic checkpoint loading.
  \end{itemize}
\end{frame}

\begin{frame}{Reproduction (semantic non-regression)}
  \begin{itemize}
    \item Re-ran the \emph{original trained checkpoints} on the modern stack
          (latest torch / PyG / numpy$\geq$2 / gymnasium).
    \item MRIR matches the committed logs \textbf{exactly}:
  \end{itemize}
  \begin{center}\small
  \begin{tabular}{lcc}
    \toprule Model & MRIR (median) & committed \\\midrule
    Graph-Q-SAT & 1.833 & 1.833 \\
    GAT-Q-SAT   & 1.667 & 1.667 \\\bottomrule
  \end{tabular}
  \end{center}
  \begin{itemize}
    \item Drop-in changes only (scatter, env construction, GATConv key remap) ---
          \textbf{semantics preserved}. Figures regenerated from the logs (no Sheets).
  \end{itemize}
\end{frame}

\begin{frame}{Results --- the central finding}
  \begin{columns}
    \begin{column}{0.58\textwidth}\centering
      \includegraphics[width=\textwidth]{thesis.png}
    \end{column}
    \begin{column}{0.42\textwidth}
      Clean attention on/off ablation, matched training sets (mean MRIR, cap 500):
      \begin{itemize}
        \item \textbf{Colouring} (in-dist): GAT \textbf{2.97} vs 2.04.
        \item \textbf{Transfer} random$\to$colouring: GAT \textbf{1.53} vs 0.94
              (Graph-Q-SAT $<1$, i.e.\ worse than MiniSat!).
        \item \textbf{Random} 3-SAT: attention \emph{does not} help (1.52 vs 2.19).
      \end{itemize}
    \end{column}
  \end{columns}
  \vspace{1mm}
  \centerline{\footnotesize \textbf{Attention helps on structured problems, not on uniform-random SAT.}}
\end{frame}

% ================================================================== Conclusions
\begin{frame}{Where the field went \& conclusions}
  \begin{itemize}
    \item Lineage: NeuroSAT $\to$ NeuroCore $\to$ NeuroComb $\to$ \textbf{NeuroBack}
          (ICLR'24): a Graph \emph{Transformer} with self-attention, \emph{offline}
          one-shot inference $\Rightarrow$ first wall-clock speed-ups on Kissat.
    \item This \textbf{validates} the attention direction (GAT-Q-SAT) and addresses
          Graph-Q-SAT's known limit (online inference too slow).
    \item \textbf{Our results}: graph attention helps on structured problems
          (colouring), in-distribution and under transfer, \emph{growing with size};
          not on random SAT. The fixed-size CNN (AlphaZeroSAT) cannot generalise.
  \end{itemize}
